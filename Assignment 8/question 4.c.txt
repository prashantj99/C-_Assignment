#include <stdio.h>

int main(void) {
	/*
	    *
    	**
    	***
    	****
    	*****
	*/
	int i=1 ,j=1;
	int rows ,cols;
	
	scanf("%d %d",&rows ,&cols);
	
	for(int i=1 ; i<=rows; i++)
	{
	    for(int j=1 ;j<=cols; j++)
	    {
	        if(j>=i)
	        {
	            printf("*");
	        }
	        else
	        {
	            printf(" ");
	        }
	        
	    }
	    
	    printf("\n");
	
	}
	
	return 0;
}
